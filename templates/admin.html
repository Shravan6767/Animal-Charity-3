async function loadContacts() { const status = document.getElementById("status"); try { // We use the exact path that worked in your browser const res = await fetch("/api/admin/contacts"); if (!res.ok) throw new Error("Server returned 404 or 500"); const data = await res.json(); const table = document.getElementById("contactsTable"); // Success! Clear the error message if (status) status.innerText = ""; data.forEach(c => { const row = table.insertRow(); row.insertCell(0).innerText = c.name || "N/A"; row.insertCell(1).innerText = c.email || "N/A"; row.insertCell(2).innerText = c.message || "N/A"; }); } catch (err) { console.error("Fetch error:", err); if (status) status.innerText = "Server not running (Check F12)"; } } async function loadDonations() { try { // FIXED: No more 127.0.0.1. Using relative path. const res = await fetch("/api/admin/donations"); if (!res.ok) return; // Silent fail if donations don't exist yet const data = await res.json(); const table = document.getElementById("donationsTable"); data.forEach(d => { const row = table.insertRow(); row.insertCell(0).innerText = d.name || "N/A"; row.insertCell(1).innerText = d.email || "N/A"; row.insertCell(2).innerText = d.amount || "0"; row.insertCell(3).innerText = d.message || "N/A"; }); } catch (err) { console.log("Donations not found or empty."); } } // Start the loading process loadContacts(); loadDonations();
